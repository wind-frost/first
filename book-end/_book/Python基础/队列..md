Queue() 创建一个空的队列
enqueue(item) 往队列中添加一个item元素
dequeue() 从队列头部删除一个元素
is_empty() 判断一个队列是否为空
size() 返回队列的大小


```
class Queue(object):
    """队列"""
    def __init__(self):
        """创建一个空的队列"""
        self.queue = []

    def enqueue(self,item):
        """往队列中添加一个item元素"""
        self.queue.append(item)

    def dequeue(self):
        """从队列头部删除一个元素"""
        return self.queue.pop(0)

    def is_empty(self):
        """判断一个队列是否为空"""
        return self.queue == []

    def size(self):
        """返回队列的大小"""
        return len(self.queue)

if __name__ == '__main__':
    q = Queue()
    print(q.size())
    print(q.is_empty())
    for i in range(5):
        q.enqueue(i)
    print(q.queue)
    print(q.size())
    print(q.is_empty())
    while True:
        try:
            print(q.dequeue())
        except IndexError:
            break
```
