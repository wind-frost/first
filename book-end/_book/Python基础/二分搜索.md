def binary_search(alist,item):
    # 计算序列长度
    n = len(alist)
    # 构造出始索引
    first = 0
    # 构造结束索引
    last = n-1
    # 构造循环查找条件
    while first <= last :
        # 构造折半查找的中间值
        mid = (first+last)//2
        # 如果元素等于中间值的情况
        if alist[mid] == item:
            return True
        # 如果元素小于中间值的情况
        elif alist[mid] > item :
            last = mid - 1
            # 如果元素大于中间值的情况
        else:
            first = mid + 1
         # 如果整个序列中查找一遍都没有找到，则返回False
    else:
        return False

if __name__ == '__main__':
    alist = [1,2,3,4,5,6,7,8,89,100,129,245]
    print(binary_search(alist,89))      # True

# 递归方法
def binary_search2(alist,item):
    if len(alist) == 0:     #判断列表是否为空，如果为空，则找不到
        return False
    else:       # 列表不为空时
        mid = len(alist)//2     # 设置中间值索引
        if item == alist[mid]:  # 如果元素等于中间值
            return True
        else:     #元素不等于中间值
            if item > alist[mid]:   # 元素大于中间值
                return binary_search2(alist[mid+1:],item)
            else:       # 元素小于中间值
                return binary_search2(alist[:mid],item)

if __name__ == '__main__':
    alist = [1,2,3,4,5,67,78,899]
    print(binary_search2(alist,899))    # True
