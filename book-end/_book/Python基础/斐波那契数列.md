# 迭代器实现
class FeiBoNaQi(object):

    def __init__(self,n):
        self.n = n
        self.num1 = 0
        self.num2 = 1
        self.current = 0

    def __iter__(self):
        return self

    def __next__(self):
        if self.current < self.n:
            num = self.num1
            self.num1,self.num2 = self.num2,self.num1+self.num2
            self.current += 1
            return num
        else:
            raise StopIteration

if __name__ == '__main__':
    fei = FeiBoNaQi(10)
    #1
    for i in fei:       # 迭代器只能迭代一次
         print(i)
     #2
    print(list(fei))
    #3
    print(tuple(fei))
    #4
    while True:
        try:
            print(next(fei))
        except StopIteration:
            break

# 生成器实现
def fei(n):
    num1,num2 = 0,1
    current = 0
    while current < n:
        num = num1
        num1,num2 = num2,num1+num2
        current += 1
        yield num

if __name__ == '__main__':
    f = fei(10)
    for i in f :

        print(i,end=" ")

    for i in fei(10):
        print(i,end=" ")
