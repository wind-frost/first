# 冒泡排序
def bubble_sort(alist):
    """冒泡排序"""
    n = len(alist)
    # 设置外层循环，构造内层循环比较条件
    for i in range(n-1):
        # 构造内层循环比较环境
        for j in range(n-1-i):
            # 如果左边元素值大于右边元素值，则交换变量值
            if alist[j] > alist[j+1] :
                # 交换变量值
                alist[j],alist[j+1] = alist[j+1],alist[j]

if __name__ == '__main__':
    alist = [23,45,1,56,23,12,45,23,0,235]
    bubble_sort(alist)
    print(alist)
