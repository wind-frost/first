"""
首先在未排序序列中找到最小（大）元素，存放到排序序列的起始位置
然后，再从剩余未排序元素中继续寻找最小（大）元素，
然后放到已排序序列的末尾。
以此类推，直到所有元素均排序完毕。
"""
# 降序
def select_sort(alist):
    n = len(alist)
    for i in range(n-1):
        # 给min_index设置默认值
        min_index = i
        # 遍历无序序列
        for j in range(i+1,n):
            # 判断当前数值是否小于最小值，如果小于，则记录索引
            if alist[min_index] < alist[j] :
                # 记录索引
                min_index = j
        # 判断min_index索引是否相同，不相同，做数值交换
        if min_index != i :
            # 不相同，做数值交换
            alist[i],alist[min_index] = alist[min_index],alist[i]

if __name__ == '__main__':
    alist = [4,56,5,2,6,8789,23,56,0,-2]
    select_sort(alist)
    print(alist)    #  [8789, 56, 56, 23, 6, 5, 4, 2, 0, -2]
