class DieDaiQi(object):
    """迭代器"""

    def __init__(self,my_list):
        """属性的初始化和赋值"""
        self.my_list = my_list
        self.current = 0 # 记录位置

    def __iter__(self):
        return self

    def __next__(self):
        """获取下一个位置元素"""
        if self.current < len(self.my_list) :
            item = self.my_list[self.current]
            self.current += 1
            return item
        else:
            raise StopIteration

if __name__ == '__main__':
    my_list = [1,2,3,4,5,6]
    die = DieDaiQi(my_list)
    for i in die:
        print(i,end=" ")
        
    while True:
        try:
            print(next(die))
        except StopIteration:
            break
